pipeline {
   agent any

   stages {
      stage('Build') {
          steps {
	    bat """
	    cd psi-project
            mvn clean install
	    """ 
          }
      }
      stage('Package') {
          steps {
            bat """
	    cd psi-project
            mvn package
	    """ 
          }
      }
      stage('Test') {
          steps {
            bat 'java -cp psi-project/target/psi-project-1.0-SNAPSHOT.jar com.sapient.App'  
          }
      }
   }
}
